    plecewise function/cases &
    \[
        a_k = \begin{cases}
            k & \text{for $k \le n/2$}\\
            n & \text{for $k=n/2$}\\
            k-1 & \text{otherwise}
        \end{cases}
    \] &
    \begin{verbatim}
a_k = \begin{cases}
    k & \text{for $k \le n/2$}\\

multiline equations(aligned at \verb|&|) &
\[
    \begin{split}\tan^2 x 
         = \sin^2 x/\cos^2 x\\
         = 1/\cos^2 x - 1
    \end{split}
\]&
\begin{verbatim}
\begin{split}\tan^2 x 
    &= \sin^2 x/\cos^2 x\\
    &= 1/\cos^2 x - 1
\end{split}
\end{verbatim}\\\hline
Systems of equations &
\[
    \left\{\begin{array}{l}
        ax+by=r_1\\
        cx+dy=r_2
    \end{array}\right.
\] &
\begin{verbatim}
\left\{\begin{array}{l}
    ax+by=r_1\\
    cx+dy=r_2
\end{array}\right.
\end{verbatim}
\end{longtable}
